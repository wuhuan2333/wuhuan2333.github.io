package demo;

class Member {
private long mid;
private String name;

public Member(long mid ,String name) {
    this.mid = mid;
    this.name = name;
}

​ public String getInfo() {
​ return “用户id：” + this.mid + “、用户姓名:” + this.name;
​ }

​ public Role roles[];

​ public void setRoles(Role roles[]) {
​ this.roles = roles;
​ }

​ public Role[] getRoles() {
​ return this.roles;
​ }
}

class Role {
private long rid;
private String name;
private Member members[];
private Quanx quanx[];

public Role(long rid ,String name) {
    this.rid = rid;
    this.name = name;
}

​ public String getInfo() {
​ return “角色id：” + this.rid + “、角色姓名:” + this.name;
​ }
​
​ public void setMembers(Member members[]) {
​ this.members = members;
​ }
​
​ public Member[] getMembers() {
​ return this.members;
​ }
​ public void setQuanx(Quanx quanx[]) {
​ this.quanx = quanx;
​ }
​ public Quanx []getQuanx() {
​ return this.quanx;
​ }
}

class Quanx {
private long qid;
private String name;
private Role roles;
public Quanx(long qid ,String name) {
this.qid = qid;
this.name = name;
}

​ public String getInfo() {
​ return “权限id：” + this.qid + “、权限姓名:” + this.name;

​ }
​
​ public void setRoles(Role roles) {
​ this.roles = roles;

public class demo {
public static void main(String args[]) {
//new用户
Member memA = new Member(1L,”用户A”);
Member memB = new Member(2L,”用户B”);
Member memC = new Member(3L,”用户C”);
//new角色
Role roleA = new Role(1L,”角色A”);
Role roleB = new Role(2L,”角色B”);
Role roleC = new Role(2L,”角色C”);
//new权限
Quanx quanxA1 = new Quanx(1001L,”权限A1”);
Quanx quanxA2 = new Quanx(1002L,”权限A2”);
Quanx quanxB1 = new Quanx(2001L,”权限B1”);
Quanx quanxB2 = new Quanx(2002L,”权限B2”);
Quanx quanxB3 = new Quanx(2003L,”权限B3”);
Quanx quanxC1 = new Quanx(3001L,”权限C1”);
Quanx quanxC2 = new Quanx(3002L,”权限C2”);
//用户赋予角色
memA.setRoles(new Role[]{roleA,roleB});
memB.setRoles(new Role[] {roleB,roleC});
memC.setRoles(new Role[] {roleA,roleC});
//角色使用权限
roleA.setQuanx(new Quanx[] {quanxA1,quanxA2});
roleB.setQuanx(new Quanx[] {quanxB1,quanxB2,quanxB3});
roleC.setQuanx(new Quanx[] {quanxC1,quanxC2});
//角色赋予用户
roleA.setMembers(new Member[] {memA,memB});
roleC.setMembers(new Member[] {memB,memC});
roleB.setMembers(new Member[] {memA,memB});
//权限给角色
quanxA1.setRoles(roleA);
quanxA2.setRoles(roleA);
quanxB1.setRoles(roleB);
quanxB2.setRoles(roleB);
quanxB3.setRoles(roleB);
quanxC1.setRoles(roleC);
quanxC2.setRoles(roleC);
System.out.println(“———根据用户找角色————–”);
System.out.println(memA.getInfo());
for(int x = 0;x<memA.getRoles().length;x++) {
System.out.println(“\t-|所拥有角色 “+memA.getRoles()[x].getInfo());
for(int y = 0;y<memA.getRoles()[x].getQuanx().length;y++) {
System.out.println(“\t\t-|角色所拥有的权限 “ + memA.getRoles()[x].getQuanx()[y].getInfo());
}
}
System.out.println(“———根据角色找权限————–”);
System.out.println(roleA.getInfo());
for(int x = 0;x<roleA.getMembers().length;x++) {
System.out.println(“\t-|拥有该角色的用户：”+roleA.getMembers()[x].getInfo());
}
for(int x = 0;x<roleA.getQuanx().length;x++) {
System.out.println(“\t-|该角色所拥有的权限” + roleA.getQuanx()[x].getInfo());
}
System.out.println(“———根据权限找用户所具备的信息————–”);
System.out.println(quanxA1.getInfo());
for(int x = 0;x<quanxA1.getRoles().getMembers().length;x++) {
System.out.println(“\t-|可以使用该权限的用户”+quanxA1.getRoles().getMembers()[x].getInfo());
}

}

}

总结

​ 在开发中遇到多对多映射的情况分为一下几个步骤行动：

​ 1.先写主要的类，确定类中的基本属性

​ 2.根据表中多对多的关系，在类中补充属性

​ 3.根据结构设置对象数据

​ 4.根据要求获取数据
